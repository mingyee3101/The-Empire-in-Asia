	option = {
		name = uic.1.a
		ai_chance = {
			factor = 100
		}

		JAP = {
			country_event = { id = uic.2 }
		}
		CHI = {
			country_event = { id = uic.2 }
		}
		SOV = {

		}
		FRA = {

		}
		ENG = {
			country_event = { id = uic.2 }
		}
		GER = {
			country_event = { id = uic.2 }
		}
		ITA = {
			country_event = { id = uic.2 }
		}
		USA = {
			country_event = { id = uic.2 }
		}
		if = {
			limit = {
				has_country_flag = UIC_flood_preparedness_cf
			}
			add_dynamic_modifier = { idea = UIC_Indochina_flood days = 240 }
			add_to_variable = { UIC_Indochina_flood_consumer_goods_factor = 0.2 }
			add_to_variable = { UIC_Indochina_flood_monthly_population = -0.3 }
			add_to_variable = { UIC_Indochina_flood_production_speed_buildings_factor = -0.3 }
			add_to_variable = { UIC_Indochina_flood_industrial_capacity_factory = -0.3 }
			add_to_variable = { UIC_Indochina_stability_factor = -0.1 }
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = UIC_flood_preparedness_cf
				}
			}
			add_dynamic_modifier = { idea = UIC_Indochina_flood days = 360 }
			add_to_variable = { UIC_Indochina_flood_consumer_goods_factor = 0.3 }
			add_to_variable = { UIC_Indochina_flood_monthly_population = -0.5 }
			add_to_variable = { UIC_Indochina_flood_production_speed_buildings_factor = -0.5 }
			add_to_variable = { UIC_Indochina_flood_industrial_capacity_factory = -0.5 }
			add_to_variable = { UIC_Indochina_stability_factor = -0.3 }
		}
		hidden_effect = {
			set_country_flag = UIC_flood_occurs
		}
	}
}

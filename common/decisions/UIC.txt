		icon = generic_political_rally

		allowed = { 
			original_tag = UIC
		}

		available = {
			
		}

		cost = 250

		fire_only_once = yes

		days_remove = 200

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 5
				check_variable = { UIC_monarchist_loyalty = 0 compare = less_than_or_equals }
			}
			modifier = {
				factor = 25
				check_variable = { UIC_monarchist_loyalty = -1 compare = less_than_or_equals }
			}
		}

		visible = {
			
		}

		complete_effect = {
			if = {
				limit = {
					has_government = communism
				}
				set_temp_variable = { UIC_befriend_monarchists_variable = party_popularity@communism }
				divide_temp_variable = { UIC_befriend_monarchists_variable = 3 }
			}
			if = {
				limit = {
					has_government = democratic
				}
				set_temp_variable = { UIC_befriend_monarchists_variable = party_popularity@democratic }
				divide_temp_variable = { UIC_befriend_monarchists_variable = 3 }
			}
			if = {
				limit = {
					has_government = fascism
				}
				set_temp_variable = { UIC_befriend_monarchists_variable = party_popularity@fascism }
				divide_temp_variable = { UIC_befriend_monarchists_variable = 3 }
			}
			multiply_temp_variable = { UIC_befriend_monarchists_variable = -1.00 }
			add_stability = var:UIC_befriend_monarchists_variable
			add_war_support = var:UIC_befriend_monarchists_variable
		}

		remove_effect = {
			set_temp_variable = { UIC_allied_monarchists_variable = party_popularity@neutrality }
			divide_temp_variable = { UIC_allied_monarchists_variable = 2 }
			if = {
				limit = {
					neutrality > 0.5
				}
				divide_temp_variable = { UIC_allied_monarchists_variable = 3 }
			}
			add_stability = var:UIC_allied_monarchists_variable
			add_war_support = var:UIC_allied_monarchists_variable
			effect_tooltip = { add_ideas = UIC_friendly_monarchists }
			hidden_effect = {
				set_variable = { UIC_monarchist_loyalty = 1 }
				UIC_political_instability_update_effect = yes
			}
		}
	}
